dat=dict()
for ff in files:
    nb=open(ff);
    nb=json.load(nb);
    dat[ff[5:11]]=nb
    if (array(nb['cells'])[array(list(map(lambda x:x['cell_type']=='code',nb['cells'])))]).any():
        dat[ff[5:11]]=array(nb['cells'])[array(list(map(lambda x:x['cell_type']=='code',nb['cells'])))]
        
aldat=dict()
for nept in dat:
    tmp=''
    for  src in dat[nept]:
        if (src['source']):                    
            tmp+=''.join(src['source'])
    aldat[nept]=tmp

Az albbi brn azt vizsgljuk meg, hogy a megrt kd hosszsgnak a fggvnyben hny if, elif s else szerepelt az adott notebookban.

lblif='if';

    plot(len(aldat[nept]),aldat[nept].count(' if'),'o',
         markerfacecolor='blue',markeredgecolor='blue',
        ms=15,label=lblif);lblif='_if';
    plot(len(aldat[nept]),aldat[nept].count('elif'),'^',

    plot(len(aldat[nept]),aldat[nept].count('else'),'s',
