    tmp=''
    for  src in dat[nept]:
        if (src['source']):                    
            tmp+=''.join(src['source'])

edgelist=[]
for i in aldat.keys():
    for j in aldat.keys():
        if i!=j:
            edgelist.append([i,j,{"weight":(fuzz.ratio(aldat[i],aldat[j])/100)**1.5}])

A networkx csomag segtsgvel megrajzolunk egy hasonlsgi grfot.

In [5]:
figsize(2*6.4,2*4.8)
G=nx.from_edgelist(edgelist)
pos=nx.spring_layout(G,k=.2)
weights = [G[u][v]['weight']*20 for u,v in G.edges()]
labels={key:key for key in G.nodes()}
nx.draw_networkx_nodes(G,pos,alpha=0.7,node_size=5000,node_color='darkred',linewidths=0)
nx.draw_networkx_labels(G,pos)
nx.draw_networkx_edges(G,pos,width=weights,alpha=0.9,edge_color='teal');
axis('off');
autoscale(True,axis='both',tight=True)
