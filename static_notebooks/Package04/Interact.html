t=linspace(0,2*pi,100);
def oszci(A,omega,phi,szin):
    plot(t,A*sin(omega*t+phi),color=szin)
    plot(pi,A*sin(omega*pi+phi),'o')
    xlim(0,2*pi)
    ylim(-3,3)
    xlabel('$t$',fontsize=20)
    ylabel(r'$A\,\sin(\omega t+\varphi)$',fontsize=20)
    grid(True)
